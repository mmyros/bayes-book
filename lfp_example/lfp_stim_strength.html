df = []
for slope in np.linspace(4, 40, 4):
    df1 = generate_fake_lfp(mouse_response_slope=slope)[0]
    df1['stim_strength'] = slope
    df.append(df1)
df = pd.concat(df)

BayesWindow(df=df, y='isi', treatment='stim_strength', group='mouse', detail='i_trial').data_box_detail()
